```
name = "paizabird"
sta = [100, 200, 50, 100, 30, 60]

# スプラット演算子なしの場合
result1 = [name, sta]
# 結果: ["paizabird", [100, 200, 50, 100, 30, 60]]
# → 配列の中に配列が入っている（入れ子構造）

# スプラット演算子ありの場合
result2 = [name, *sta]
# 結果: ["paizabird", 100, 200, 50, 100, 30, 60]
# → staの中身が展開されて、一つの配列になる
```
