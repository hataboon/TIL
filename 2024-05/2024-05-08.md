N = gets.to_i
total = 0

N.times do 
  answer, result = gets.split
  if answer.size != result.size
      total += 0
  else    
      diff_count = 0   
      
      answer.size.times do |i|
         diff_count += 1 if answer[i] != result[i]
      end
      
      if diff_count == 0 
          total += 2 
      elsif diff_count == 1
          total += 1
      else
          total += 0
      end
  end
end

puts total
